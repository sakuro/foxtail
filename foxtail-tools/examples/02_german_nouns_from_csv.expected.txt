-dog =
    { $count ->
        [one]
            { $case ->
               *[nominative] Hund
                [accusative] Hund
                [dative] Hund
                [genitive] Hundes
            }
       *[other]
            { $case ->
               *[nominative] Hunde
                [accusative] Hunde
                [dative] Hunden
                [genitive] Hunde
            }
    }
    .gender = masculine
